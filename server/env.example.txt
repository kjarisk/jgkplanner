# JGK Planner - Server Environment Variables
# Copy this file to .env and fill in your values

# ============================================
# REQUIRED - Server Configuration
# ============================================

# Environment: development or production
NODE_ENV=development

# Server port (default: 3001)
PORT=3001

# ============================================
# REQUIRED - Security Secrets
# ============================================
# IMPORTANT: Generate secure random strings for production!
# Use: openssl rand -base64 32

# JWT secret for token signing (min 32 characters)
JWT_SECRET=your-secure-jwt-secret-change-this-in-production

# Session secret for express-session (min 32 characters)
SESSION_SECRET=your-secure-session-secret-change-this-in-production

# ============================================
# REQUIRED - URLs
# ============================================

# Frontend URL (for CORS and OAuth redirects)
CLIENT_URL=http://localhost:5173

# Backend URL (for OAuth callbacks)
SERVER_URL=http://localhost:3001

# ============================================
# OPTIONAL - OAuth Providers
# ============================================
# Get these from the respective developer consoles:
# Google: https://console.cloud.google.com/apis/credentials
# Facebook: https://developers.facebook.com/apps/

# Google OAuth
GOOGLE_CLIENT_ID=
GOOGLE_CLIENT_SECRET=

# Facebook OAuth
FACEBOOK_APP_ID=
FACEBOOK_APP_SECRET=

# ============================================
# PRODUCTION CHECKLIST
# ============================================
# Before deploying to production, ensure:
# 1. NODE_ENV=production
# 2. JWT_SECRET is a secure random string
# 3. SESSION_SECRET is a secure random string
# 4. CLIENT_URL points to your production frontend
# 5. SERVER_URL points to your production backend
# 6. OAuth credentials are set up for production domains
# 7. HTTPS is enabled (required for OAuth and secure cookies)
